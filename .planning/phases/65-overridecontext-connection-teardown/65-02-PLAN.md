```cpp
// Registration helper signature (extern "C" — called from Rust via libduckdb-sys::duckdb_database)
bool sv_register_table_function(
    duckdb_database db_handle,
    const char *name,
    const sv_table_function_signature_t *sig,   // argument LogicalType ids + flags
    sv_bind_fn_t   rust_bind,                   // Rust bind function pointer
    sv_init_fn_t   rust_init,                   // Rust init function pointer
    sv_execute_fn_t rust_execute);              // Rust execute function pointer

// SvRustExtraInfo : public TableFunctionInfo
// Holds the three Rust function pointers + the function name C-string.
// NO duckdb_connection / duckdb_database stashed here.

// Trampoline shape (C++):
static unique_ptr<FunctionData> sv_trampoline_bind(
    ClientContext &context,
    TableFunctionBindInput &input,
    vector<LogicalType> &return_types,
    vector<string> &names) {
    auto &extra = input.info->Cast<SvRustExtraInfo>();
    Connection probe(*context.db);  // empirically validated, see 65-READ-PATH-SPIKE.md
    auto conn = reinterpret_cast<duckdb_connection>(&probe);
    void *rust_bind_data = nullptr;
    char *error_msg = nullptr;
    bool ok = extra.rust_bind(conn, &input, &return_types, &names,
                              &rust_bind_data, &error_msg);
    // probe destructs here
    if (!ok) { throw BinderException(...); }
    return make_uniq<SvRustBindData>(rust_bind_data, extra.rust_drop_bind_data);
}
```
