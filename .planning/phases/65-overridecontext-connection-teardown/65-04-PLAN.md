    ```
    #[test]
    fn override_context_carries_no_long_lived_connection() {
        // Phase 65 RESEARCH ┬¦6: OverrideContext must not carry a duckdb_connection
        // or a CatalogReader (which itself used to carry a duckdb_connection). If
        // either reappears, the in-process RWŌåÆRO busy-spin returns. This test is
        // the structural regression guard for that invariant.
        let parse_rs = std::fs::read_to_string(
            concat!(env!("CARGO_MANIFEST_DIR"), "/src/parse.rs"),
        )
        .expect("read src/parse.rs");

        // Locate the PHASE-65-GUARD marker (inside the struct body); scan from the
        // marker to the next unmatched `}` to obtain exactly the struct body. This
        // is robust against the struct moving in the file or growing in size.
        let marker = "// PHASE-65-GUARD: do not reintroduce duckdb_connection or CatalogReader field here.";
        let marker_pos = parse_rs.find(marker)
            .expect("PHASE-65-GUARD marker missing in src/parse.rs ŌĆö Plan 02 Task 1 / Plan 04 Task 1 must add it");

        // Brace-depth scan from marker outwards: rewind to the most recent `{` (the
        // struct's opening brace, depth 1) and forward-scan tracking depth until
        // depth returns to 0.
        let prefix = &parse_rs[..marker_pos];
        let open_brace = prefix.rfind('{')
            .expect("opening brace of OverrideContext not found before marker");
        let mut depth: i32 = 0;
        let bytes = parse_rs.as_bytes();
        let mut end = open_brace;
        for (i, &b) in bytes[open_brace..].iter().enumerate() {
            match b {
                b'{' => depth += 1,
                b'}' => {
                    depth -= 1;
                    if depth == 0 {
                        end = open_brace + i + 1;
                        break;
                    }
                }
                _ => {}
            }
        }
        assert!(end > open_brace, "could not locate closing brace of OverrideContext");
        let block = &parse_rs[open_brace..end];

        assert!(
            !block.contains("conn: ffi::duckdb_connection")
                && !block.contains("conn: libduckdb_sys::duckdb_connection"),
            "OverrideContext re-acquired a duckdb_connection field ŌĆö see Phase 65 RESEARCH ┬¦6"
        );
        assert!(
            !block.contains("catalog: crate::catalog::CatalogReader")
                && !block.contains("catalog: CatalogReader"),
            "OverrideContext re-acquired a CatalogReader field ŌĆö see Phase 65 RESEARCH ┬¦6"
        );
        // Positive assertion: the new field must be present.
        assert!(
            block.contains("db_handle"),
            "OverrideContext missing the db_handle field ŌĆö see Phase 65 RESEARCH ┬¦6"
        );
    }
    ```
