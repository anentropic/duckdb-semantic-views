/// FFI entrypoint matching `SvBindFn` signature.
/// Called by C++ sv_trampoline_bind with a per-call `conn` pointer.
#[unsafe(no_mangle)]
pub unsafe extern "C" fn sv_list_semantic_views_bind(
    conn: ffi::duckdb_connection,
    bind_input: *mut c_void,
    return_types: *mut c_void,
    names: *mut c_void,
    bind_data_out: *mut *mut c_void,
    error_msg_out: *mut *mut c_char,
) -> bool {
    std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
        // Probe catalog presence per-call
        let catalog_table_present = unsafe { probe_catalog_present(conn) };
        let reader = CatalogReader::new(conn, catalog_table_present);

        // Run existing bind body (the same logic that was in the
        // duckdb-rs VTab::bind method) against the per-call reader.
        // Populate return_types / names; build a Rust Box for bind data.
        let entries = match reader.list_all() {
            Ok(e) => e,
            Err(msg) => {
                // Return Rust-allocated CString via error_msg_out
                let cstr = std::ffi::CString::new(msg).unwrap();
                unsafe { *error_msg_out = cstr.into_raw(); }
                return false;
            }
        };

        // Populate names + return_types via the C++ helper accessors that
        // wrap vector<string>& / vector<LogicalType>&. Helper functions:
        //   sv_push_bind_name(names, name_cstr)
        //   sv_push_bind_return_type(return_types, type_id)
        // (Add to cpp/src/shim.cpp + extern in src/lib.rs if not present)
        ...

        // Box the bind data for transport across FFI
        let bind_data = Box::new(ListSemanticViewsBindData { entries });
        unsafe { *bind_data_out = Box::into_raw(bind_data) as *mut c_void; }
        true
    })).unwrap_or(false)
}

#[unsafe(no_mangle)]
pub unsafe extern "C" fn sv_list_semantic_views_init(
    bind_data: *mut c_void,
    init_input: *mut c_void,
    init_data_out: *mut *mut c_void,
    error_msg_out: *mut *mut c_char,

pub unsafe extern "C" fn sv_list_semantic_views_execute(
    bind_data: *mut c_void,
    init_data: *mut c_void,
    output: *mut c_void,
    error_msg_out: *mut *mut c_char,
) -> bool { /* iterate entries; write into output DataChunk */ }

#[unsafe(no_mangle)]
pub unsafe extern "C" fn sv_drop_list_semantic_views_bind_data(p: *mut c_void) {
    if !p.is_null() {
        let _ = unsafe { Box::from_raw(p as *mut ListSemanticViewsBindData) };
    }

Body reuse: the bind logic itself (the `reader.list_all()` call, the column-name + type construction) is unchanged from the existing duckdb-rs VTab::bind code. Only the SHAPE of the bind callback changes — FFI instead of trait method, raw pointers instead of typed BindInfo wrappers. The execute body similarly reuses the existing iteration logic, but writes to a raw output DataChunk pointer instead of a duckdb-rs WritableVector.
