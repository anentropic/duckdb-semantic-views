    /// Phase 62 — `ParseError::position` MUST be a byte offset into the
    /// user's input string, NOT into the rewritten SQL. RESEARCH.md Q1
    /// proves this is already true; this proptest pins it as a contract
    /// across arbitrary leading whitespace + comments.
    #[test]
    fn parse_error_position_is_byte_offset_into_user_input_smoke() {
        // Smoke test (proptest variant below). The malformed CREATE has a
        // bad token at byte offset N; prepending whitespace shifts the
        // reported position by exactly that many bytes.
        use semantic_views::parse::validate_and_rewrite;

        let canonical = "CREATE SEMANTIC VIEW v AS TBLES (t);";
        let bad_token_offset = canonical.find("TBLES").unwrap();

        let err = validate_and_rewrite(canonical).expect_err("should fail");
        let pos = err.position.expect("position should be set for clause typo");
        assert_eq!(pos, bad_token_offset, "smoke: canonical case");

        // Whitespace prefix shifts the position by its byte length.
        let prefixed = format!("   \n\t{canonical}");
        let prefix_len = prefixed.len() - canonical.len();
        let err2 = validate_and_rewrite(&prefixed).expect_err("should fail");
        let pos2 = err2.position.expect("position should be set");
        assert_eq!(pos2, prefix_len + bad_token_offset, "smoke: with whitespace");

        // Line-comment prefix.
        let with_comment = format!("-- a comment\n{canonical}");
        let comment_len = with_comment.len() - canonical.len();
        let err3 = validate_and_rewrite(&with_comment).expect_err("should fail");
        let pos3 = err3.position.expect("position should be set");
        assert_eq!(pos3, comment_len + bad_token_offset, "smoke: with line comment");
    }
