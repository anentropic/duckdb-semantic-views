```rust
//! Phase 65.1 Plan 02 — WR-02 D-08/D-09 + CR-02 D-05.
//! Verifies the error_buf ABI surfaces underlying registration failures.

use std::ffi::{CStr, CString};
use std::os::raw::c_char;
use std::ptr;

// Extern declaration mirrors src/lib.rs — the symbol is exported by the
// extension's static lib and resolvable at test link time when the test
// is built without the cdylib feature.
extern "C" {
    fn sv_register_table_function(
        db_handle: libduckdb_sys::duckdb_database,
        name: *const c_char,
        arg_types: *const std::ffi::c_void, // LogicalType opaque to Rust
        arg_count: usize,
        bind_cb: Option<unsafe extern "C" fn()>,
        exec_cb: Option<unsafe extern "C" fn()>,
        init_cb: Option<unsafe extern "C" fn()>,
        error_buf: *mut c_char,
        error_buf_len: usize,
    ) -> bool;
}

#[test]
fn init_extension_surfaces_registration_error_buf() {
    let mut db: libduckdb_sys::duckdb_database = ptr::null_mut();
    let rc = unsafe { libduckdb_sys::duckdb_open(ptr::null(), &mut db) };
    assert_eq!(rc, libduckdb_sys::DuckDBSuccess, "duckdb_open failed");

    let name = CString::new("test_force_fail_tf").unwrap();
    // Dummy non-null bind/exec function pointers — use the extern fn shim
    // signatures actually exported. The point is init_cb is null.
    extern "C" fn dummy() {}

    let mut error_buf = [0u8; 1024];
    let ok = unsafe {
        sv_register_table_function(
            db,
            name.as_ptr(),
            ptr::null(),
            0,
            Some(std::mem::transmute(dummy as extern "C" fn())),
            Some(std::mem::transmute(dummy as extern "C" fn())),
            None, // ← deliberately null init_cb
            error_buf.as_mut_ptr().cast::<c_char>(),
            error_buf.len(),
        )
    };

    unsafe { libduckdb_sys::duckdb_close(&mut db) };

    assert!(!ok, "registration must refuse null init_cb");
    let msg = unsafe { CStr::from_ptr(error_buf.as_ptr().cast::<c_char>()) }
        .to_string_lossy()
        .into_owned();
    assert!(
        msg.contains("init_cb is mandatory"),
        "error_buf must contain `init_cb is mandatory` substring; got: {msg}"
    );
}
```

| Caller of `init_extension` | DuckDB host calls our extension's init hook; surfaces our error message verbatim in `LOAD semantic_views` failure |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-65.1-02-01 | Information disclosure | `sv_register_*` stderr writes | mitigate | Replace `fprintf(stderr, ...)` with `snprintf(error_buf, ...)` — error stays in the caller's diagnostic channel, no leak to global stderr that may be redirected/logged elsewhere (D-09) |
| T-65.1-02-02 | Denial of service | Unbounded row stream from null init_cb | mitigate | Refuse registration at boot when `init_cb == nullptr` (D-05). Failing at LOAD is the right blast radius |
| T-65.1-02-03 | Tampering | Buffer overflow in snprintf | accept | `snprintf(buf, buf_len, ...)` truncates and NUL-terminates within buffer cap — safe by construction |
</threat_model>
